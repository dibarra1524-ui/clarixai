  <script>
  document.addEventListener('DOMContentLoaded', () => {
    // Cursor
    const cursor = document.getElementById('cursor');
    const ring = document.getElementById('cursor-ring');
    document.addEventListener('mousemove', e => {
      if (cursor) {
        cursor.style.left = e.clientX + 'px';
        cursor.style.top = e.clientY + 'px';
      }
      if (ring) {
        ring.style.left = e.clientX + 'px';
        ring.style.top = e.clientY + 'px';
      }
    });

    // Chat test to proxy
    const chatMessages = document.getElementById('chat-messages');
    const chatInput = document.getElementById('chat-input');
    const chatSend = document.getElementById('chat-send');

    if (chatSend && chatInput && chatMessages) {
      chatSend.addEventListener('click', async () => {
        const text = chatInput.value.trim();
        if (!text) return;
        chatMessages.innerHTML += `<div class="message user">${text}</div>`;
        chatInput.value = '';
        try {
          const res = await fetch('https://clarix-ai-proxy.dibarra1524.workers.dev/', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              model: 'claude-3-5-sonnet-20241022',
              max_tokens: 200,
              messages: [{ role: 'user', content: text }]
            })
          });
          const data = await res.json();
          chatMessages.innerHTML += `<div class="message ai">${data.content?.[0]?.text || 'No response'}</div>`;
          chatMessages.scrollTop = chatMessages.scrollHeight;
        } catch (err) {
          chatMessages.innerHTML += `<div class="message ai">Error: ${err}</div>`;
        }
      });
    }

    // Nav scroll
    const nav = document.getElementById('nav');
    if (nav) {
      window.addEventListener('scroll', () => {
        nav.classList.toggle('scrolled', window.scrollY > 40);
      });
    }

    console.log("Script running - no syntax error");
  });
  </script>
